clear all; clc;



img=imread("Image3.png");
figure;
subplot(1,4,1);
imshow(img);

filter=1/16*[1 2 1;2 4 2;1 2 1];
conv=conv2(img,filter,"same");
subplot(1,4,2);
conv=uint8(conv);
imshow(conv);

row=size(conv,1);
column=size(conv,2);
thrsVal=150;



for i=1:row
    for j=1:column
        if(conv(i,j)<thrsVal)
            conv(i,j)=0;
        end
    end
end



subplot(1,4,3); 
imshow(conv);
